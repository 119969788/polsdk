# Polymarket SDK 配置
# 复制此文件为 .env 并填写实际值

# ===== 必需配置 =====
# 您的私钥（用于签名交易）
PRIVATE_KEY=your_private_key_here

# ===== 跟单配置 =====
# 要跟单的钱包地址，多个地址用逗号分隔
FOLLOW_WALLETS=0x123...,0x456...

# 最大持仓金额 (USD)
DIP_ARB_MAX_POSITION=500
